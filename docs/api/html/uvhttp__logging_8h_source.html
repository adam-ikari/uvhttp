39/* Debug 模式下，日志输出到 stderr */
40# define UVHTTP_LOG(level, ...) \
41 do { \
42 const char* level_str = "UNKNOWN"; \
43 switch (level) { \
44 case UVHTTP_LOG_LEVEL_TRACE: \
45 level_str = "TRACE"; \
46 break; \
47 case UVHTTP_LOG_LEVEL_DEBUG: \
48 level_str = "DEBUG"; \
49 break; \
50 case UVHTTP_LOG_LEVEL_INFO: \
51 level_str = "INFO"; \
52 break; \
53 case UVHTTP_LOG_LEVEL_WARN: \
54 level_str = "WARN"; \
55 break; \
56 case UVHTTP_LOG_LEVEL_ERROR: \
57 level_str = "ERROR"; \
58 break; \
59 case UVHTTP_LOG_LEVEL_FATAL: \
60 level_str = "FATAL"; \
61 break; \
62 } \
63 fprintf(stderr, "[%s] %s:%d %s(): ", level_str, __FILE__, \
64 __LINE__, __func__); \
65 fprintf(stderr, __VA_ARGS__); \
66 fprintf(stderr, "\n"); \
67 fflush(stderr); \
68 } while (0)
