58 /* 编译宏 - 执行中间件链
59  *
60  * 使用说明：
61  * 1. 此宏使用 static const 数组，只分配一次，零运行时开销
62  * 2. 自动创建上下文，生命周期与请求一致
63  * 3. 使用 goto 机制确保在任何上下文中都能正确停止执行
64  * 4. 自动调用上下文清理函数
65  * 5. 可以在任何上下文中使用（void/non-void 函数、嵌套循环等）
66  *
67  * 重要限制：
68  * - 参数 req 和 resp 不应有副作用（如函数调用），因为它们会被多次使用
69  * - 推荐用法：先获取指针，再传递给宏
70  * - 上下文在每个中间件调用后会被重置，无法在中间件间共享状态
71  * - 如果中间件返回 STOP，宏会立即跳转到标签，后续代码不会执行
72  *
73  * 错误示例：
74  * UVHTTP_EXECUTE_MIDDLEWARE(get_request(), get_response(), mw1); // ❌ 函数被多次调用
75  *
76  * 正确示例：
77  * uvhttp_request_t* req = get_request(); // ✅ 先获取指针
78  * uvhttp_response_t* resp = get_response();
79  * UVHTTP_EXECUTE_MIDDLEWARE(req, resp, mw1);
80  *
81  * 示例：
82  * void handler(uvhttp_request_t* req, uvhttp_response_t* resp) {
83  * UVHTTP_EXECUTE_MIDDLEWARE(req, resp,
84  * logging_middleware,
85  * auth_middleware,
86  * cors_middleware
87  * );
88  * // 处理请求...
89  * }
90  *
91  * 性能对比：
92  * - 宏系统：零运行时分配，~29% 性能提升
93  * - 动态系统：需要堆分配，但更灵活
94  *
95  * 选择建议：
96  * - 使用宏系统：如果中间件链固定且性能关键
97  * - 使用动态系统：如果需要灵活配置和路径匹配
98  */
99 #define UVHTTP_EXECUTE_MIDDLEWARE(req, resp, ...) \
100  do { \
101  static const uvhttp_middleware_handler_t handlers[] = { __VA_ARGS__ }; \
102  const size_t handler_count = sizeof(handlers)/sizeof(handlers[0]); \
103  uvhttp_middleware_context_t _mw_ctx = {0}; \
104  for (size_t i = 0; i < handler_count; i++) { \
105  if (handlers[i] && handlers[i](req, resp, &_mw_ctx) != UVHTTP_MIDDLEWARE_CONTINUE) { \
106  if (_mw_ctx.cleanup) _mw_ctx.cleanup(_mw_ctx.data); \
107  goto uvhttp_middleware_stop_execution; \
108  } \
109  } \
110  } while(0); \
111  uvhttp_middleware_stop_execution: (void)0
