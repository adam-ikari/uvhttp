43 /*
44  * 执行中间件链
45  *
46  * 使用示例：
47  * static int my_handler(uvhttp_request_t* req, uvhttp_response_t* resp) {
48  * UVHTTP_MIDDLEWARE(req, resp,
49  * logging_middleware,
50  * cors_middleware,
51  * auth_middleware
52  * );
53  *
54  * // 处理请求...
55  * return 0;
56  * }
57  *
58  * 特性：
59  * - 零动态分配：使用 static const 数组
60  * - 零运行时开销：编译时确定
61  * - 类型安全：使用函数指针类型
62  * - 可预测：使用 goto 而非 continue
63  *
64  * 注意：
65  * - 中间件函数必须返回 UVHTTP_MIDDLEWARE_CONTINUE 或 UVHTTP_MIDDLEWARE_STOP
66  * - 中间件函数不能有副作用（如修改全局状态）
67  * - 中间件函数必须快速执行（避免阻塞）
68  * - 不支持中间件状态（每个请求独立）
69  */
70 #define UVHTTP_MIDDLEWARE(req, resp, ...) \
71  do { \
72  static const uvhttp_middleware_fn_t _mw_handlers[] = { __VA_ARGS__ }; \
73  const size_t _mw_count = sizeof(_mw_handlers) / sizeof(_mw_handlers[0]); \
74  for (size_t _mw_i = 0; _mw_i < _mw_count; _mw_i++) { \
75  if (_mw_handlers[_mw_i] && _mw_handlers[_mw_i](req, resp) != UVHTTP_MIDDLEWARE_CONTINUE) { \
76  goto _mw_end; \
77  } \
78  } \
79  } while(0); \
80  _mw_end: (void)0
